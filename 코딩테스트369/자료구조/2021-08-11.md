class Node<T>{
    T data;
    Node<T> next = null;

    public Node(T data) {
        this.data = data;
    }
}

public class SingleLinkedList<T> {
    public Node<T> head = null;

    public void addNode(int data){
        if(head == null){
            head = new Node(data);
        }else{
            Node<T> node = this.head;
            while (node.next != null){
                node = node.next;
            }
            node.next = new Node(data);
        }
    }

    public void printAll(){
        if(head != null){
            Node<T> node = head;
            System.out.println(node.data);
            while (node.next != null){
                 node = node.next;
                 System.out.println(node.data);
            }
        }
    }

    public void addNodeInside(T data, T findData){
        Node<T> findNode = search(findData);

        if(findNode == this.head){
            findNode.next = new Node<>(data);
        }else{
            Node<T> nextNode = findNode.next;
            findNode.next = new Node<>(data);
            findNode.next.next = nextNode;
        }
    }

    private Node<T> search(T findData) {

        if(findData == head.data){
            return head;
        }else{
            Node<T> node = head;

            while (node.next != null){
                if(node.data == findData){
                    return node;
                }
                else{
                    node = node.next;
                }
            }
        }
        return null;
    }

    public boolean delNode(T findData){

        if(head == null){
            return false;
        }

        if(head.data == findData){
            head = head.next;
            return true;
        }else{
            Node<T> node = head;
            while (node.next != null){
                if(node.next.data == findData){
                    node.next = node.next.next;
                    return true;
                }
                node = node.next;
            }
        }
        return false;
    }
}

                return node;

        return null;

    public boolean insertToFront(T exitedData, T addData){
        Node<T> newNode = new Node<>(addData);

        if(head == null){
            head = newNode;
            tail = head;
            return true;

            Node<T> prevNode = head;
            head = newNode;
            newNode.next = prevNode;
